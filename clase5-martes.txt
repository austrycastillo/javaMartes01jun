package com.educacionit.modules;

public class SistemaIMC {
	public static void calcularIMC(double altura, double peso) {
		double imc = peso / (altura * altura);
		System.out.println("Tu indice de masa corporal es " + imc);
		System.out.println(recomendar(imc));
	}
	public static String recomendar(double imc) {
		String rta;
		if(imc < 18.5)
			rta="Bajo peso, te recomendamos seguir una dieta con blabla";
		else if(imc >=18.5 && imc <=24.9)
			rta= "imc normal ;)";
		else if(imc >=25 && imc <=29.9)
			rta= "tienes sobrepeso, puedes hacer blabla";
		else
			rta= "estás un poco obeso blabla..";
		
		return rta;
	}
}
